public class Main {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int a = sc.nextInt();
		int b = sc.nextInt();
		int sum = 0;
		
		for (int i = a; i < b; i++) {
			if (b-a == 1) {
				sum = a+b;
			} else{
				sum = a + b + i;
			}
		}
		System.out.println(sum);
		
	}
}
